        // Load items into a slider
        function loadSlider(sliderId, items) {
            const slider = sliders[sliderId];
            slider.innerHTML = '';

            items.forEach(item => {
                const sliderItem = document.createElement('div');
                sliderItem.className = 'slider-item';

                let discountTag = '';
                if (item.discount) {
                    discountTag = `<div class="discount-tag">-${item.discount}</div>`;
                }

                let originalPrice = '';
                if (item.originalPrice) {
                    originalPrice = `<span class="original-price">${item.originalPrice}</span>`;
                }

                // Get saved rating or use default
                const savedRating = ratings[item.id] || item;

                // Create rating stars
                let ratingStars = '<div class="rating">';
                const ratingValue = savedRating.rating || 4;
                const reviewCount = savedRating.reviews || item.reviews;

                // Calculate full and half stars
                const fullStars = Math.floor(ratingValue);
                const halfStar = ratingValue % 1 >= 0.5 ? 1 : 0;
                const emptyStars = 5 - fullStars - halfStar;

                // Add full stars
                for (let i = 0; i < fullStars; i++) {
                    ratingStars += '<i class="fas fa-star"></i>';
                }

                // Add half star
                if (halfStar) {
                    ratingStars += '<i class="fas fa-star-half-alt"></i>';
                }

                // Add empty stars
                for (let i = 0; i < emptyStars; i++) {
                    ratingStars += '<i class="far fa-star"></i>';
                }

                ratingStars += ` <span>(${reviewCount})</span></div>`;

                sliderItem.innerHTML = `
                    ${discountTag}
                    <img src="${item.image}" alt="${item.name}" class="slider-img">
                    <div class="slider-info">
                        <div class="slider-name">${item.name}</div>
                        ${ratingStars}
                        <div class="slider-price">
                            <span class="current-price">${item.price}</span>
                            ${originalPrice}
                        </div>
                    </div>
                `;

                // Add click event to show details
                sliderItem.addEventListener('click', () => showDetail(item));

                slider.appendChild(sliderItem);
            });
        }

        // Rate a product
        function rateProduct(productId) {
            // Get the current product data
            const product = fishData.recent.concat(fishData.wild, fishData.ornamental, fishData.fighting, fishData.others)
                .find(p => p.id === productId);

            // Get saved rating or initialize
            const savedRating = ratings[productId] || {...product};

            // Update rating
            if (!savedRating.rating) savedRating.rating = product.rating;
            if (!savedRating.reviews) savedRating.reviews = product.reviews;

            // Add a rating (increment by 0.1 for each like, max 5)
            savedRating.rating = Math.min(savedRating.rating + 0.1, 5);
            savedRating.reviews += 1;

            // Save to ratings object
            ratings[productId] = savedRating;

            // Save to localStorage
            localStorage.setItem('productRatings', JSON.stringify(ratings));

            // Update display
            updateRatingDisplay(productId);

            // Show animation on like button
            likeBtn.innerHTML = '<i class="fas fa-heart"></i> អ្នកចូលចិត្តរួចហើយ!';
            likeBtn.style.backgroundColor = '#e74c3c';

            setTimeout(() => {
                likeBtn.innerHTML = '<i class="fas fa-heart"></i> ចូលចិត្ត';
                likeBtn.style.backgroundColor = '';
            }, 2000);
        }
